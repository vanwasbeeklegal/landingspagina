        function calculateBenefit() {
            const salary = parseFloat(document.getElementById('calc-salary').value);
            const years = parseFloat(document.getElementById('calc-years').value);

            if (isNaN(salary) || isNaN(years) || salary <= 0 || years < 0) {
                alert("Vul a.u.b. geldige waarden in voor salaris en dienstjaren.");
                return;
            }

            // Transitievergoeding: 1/3 maandsalaris per jaar
            const baseTV = (salary * years) / 3;
            // +23% extra bonus
            const totalTV = baseTV * 1.23;

            const formattedTV = new Intl.NumberFormat('nl-NL', { style: 'currency', currency: 'EUR' }).format(totalTV);
            
            document.getElementById('res-compensation').innerText = formattedTV;
            document.getElementById('calc-result').classList.remove('hidden');
            
            document.getElementById('calc-result').scrollIntoView({behavior: 'smooth', block: 'nearest'});
        }

        async function analyzeWithAI() {
            const input = document.getElementById('ai-input').value;
            if (!input || input.length < 20) {
                alert("Voer alstublieft een representatief stuk tekst uit uw VSO in.");
                return;
            }

            const btn = document.getElementById('ai-analyze-btn');
            const container = document.getElementById('ai-result-container');
            const loading = document.getElementById('ai-loading');
            const resultText = document.getElementById('ai-result-text');

            btn.disabled = true;
            btn.innerHTML = '<span>Bezig met analyseren ✨</span>';
            container.classList.remove('hidden');
            loading.classList.remove('hidden');
            resultText.classList.add('hidden');

            const systemPrompt = "Je bent een expert in Nederlands arbeidsrecht bij Van Wasbeek Wet & Recht. Analyseer de tekst uit een vaststellingsovereenkomst. Geef 3 korte, kritische aandachtspunten in bulletpoints en sluit af met een motiverende zin om contact op te nemen voor een volledige check. Gebruik een professionele, geruststellende toon.";

            try {
                const data = await fetchWithRetry(input, systemPrompt);
                const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
                if (text) {
                    resultText.innerHTML = text.replace(/\n/g, '<br>').replace(/\*/g, '•');
                } else {
                    throw new Error("Geen resultaat");
                }
            } catch (error) {
                resultText.innerHTML = "<p class='text-red-600'>Er is een fout opgetreden bij de AI-analyse. Probeer het later opnieuw.</p>";
            } finally {
                loading.classList.add('hidden');
                resultText.classList.remove('hidden');
                btn.disabled = false;
                btn.innerHTML = '<span>Analyseer mijn VSO ✨</span>';
            }
        }

        async function fetchWithRetry(userQuery, systemPrompt) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: userQuery }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] }
            };
            let delay = 1000;
            for (let i = 0; i < 5; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) throw new Error("API Fout");
                    return await response.json();
                } catch (e) {
                    if (i === 4) throw e;
                    await new Promise(r => setTimeout(r, delay));
                    delay *= 2;
                }
            }
        }
